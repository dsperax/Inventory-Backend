/* Lógico_LOJACAR SQL: */

CREATE TABLE CATEGORIA (
    ID LONG PRIMARY KEY,
    NOME VARCHAR,
    UNIQUE (ID, NOME)
);

CREATE TABLE PRODUTO (
    ID LONG PRIMARY KEY,
    NOME VARCHAR,
    DESCRICAO VARCHAR,
    PRECO DECIMAL,
    CATEGORIA VARCHAR,
    FOREIGN KEY(CATEGORIA) REFERENCES CATEGORIAS(NOME),
    UNIQUE (ID, CATEGORIA, NOME)
);

/* Lógico_LOJACAR PLSQL: */

CREATE TABLE CATEGORIA (
    ID CLOB PRIMARY KEY,
    NOME VARCHAR2(20),
    UNIQUE (ID, NOME)
);

CREATE TABLE PRODUTO (
    ID CLOB PRIMARY KEY,
    NOME VARCHAR2(50),
    DESCRICAO VARCHAR2(300),
    PRECO NUMBER(10,0),
    CATEGORIA VARCHAR2(20),
    FOREIGN KEY(CATEGORIA) REFERENCES CATEGORIAS(NOME),
    UNIQUE (ID, CATEGORIA, NOME)
);

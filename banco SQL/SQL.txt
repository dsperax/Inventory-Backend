CREATE TABLE CATEGORIA (
    ID LONG PRIMARY KEY,
    NOME VARCHAR,
    UNIQUE (ID, NOME)
);

CREATE TABLE PRODUTO (
    ID LONG PRIMARY KEY,
    NOME VARCHAR,
    DESCRICAO VARCHAR,
    PRECO DECIMAL,
    CATEGORIA VARCHAR,
    FOREIGN KEY(CATEGORIA) REFERENCES CATEGORIAS(NOME),
    UNIQUE (ID, CATEGORIA, NOME)
);
